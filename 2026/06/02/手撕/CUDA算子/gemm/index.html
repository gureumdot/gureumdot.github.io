GEMM

CUDA GEMM 算子优化 手撕
手撕 CUDA算子
Publish Date:   2026-06-02

手撕目标

GEMM = C[M,N] = A[M,K] · B[K,N]。所有优化只围绕一个目标：

逼近"计算访存比 × Occupancy"乘积的最高点 —— 既要每搬一次数据多算几次，又要让计算单元别空转。

因为内存墙的存在，GPU 算力远超访存带宽。朴素写法（naive）一次访存对应约 1 次 FLOP，被带宽活活拖死。优化的方向是把数据从 HBM 搬到离 ALU 越来越近的存储层级（HBM → shared memory → register），每搬一次尽量多复用（拉高计算访存比）；同时保证 SM 上有足够多 warp 可调度，等数据时不空转（保住 Occupancy）。这两个量是互相拉扯的，最优参数是它们的平衡点 —— 详见第零章。

我们要手撕的版本是经过九级优化叠加的 double buffering 最终版，它把 block tile、thread tile、warp tile、向量化读取、swizzling、z-order、位运算优化、double buffer 全部融合在一起，在 4060 Ti 上达到约 cuBLAS 80% 的性能。

手撕代码（doublebuffering 最终版，逐行批注）

下面是九级优化的融合版本。模板参数的取值理由放在第零章详谈，这里先整体过一遍代码，最后去各节深挖每个优化点。

// double buffering —— 九级优化融合版（最终手撕版）
template<int Bm = 128, int Bn = 128, int Bk = 8, int blockSize = 256, int A_BLOCK_X = 8,
         int A_BLOCK_Y = 32, int B_BLOCK_X = 32, int C_BLOCK_X = 16, int C_BLOCK_Y = 16,
         int C_WARP_X = 8, int C_WARP_Y = 4, int C_WARP_DIM_X = 2, int Tm = 8, int Tn = 8>
__global__ void doublebufferingGEMM(float* A, float* B, float* C, const int M, const int K,
                                  const int N) {

  // ★ gemm2 block tile | gemm9 double buffer
  // As/Bs 各两份：buffer[0] 正在计算，buffer[1] 正在预取（或反之）
  __shared__ float As[2][Bk][Bm];  // 存储转置后的 tileA
  __shared__ float Bs[2][Bk][Bn];  // 存储 tileB

  // ★ gemm3 thread tile：每线程负责 Tm×Tn 个 C 元素，寄存器累加
  float Ct[Tm][Tn] = {0.0};

  // ★ gemm3 thread tile：寄存器缓存 A 列向量和 B 行向量（外积所用）
  // ★ gemm9 double buffer：[2] 双缓冲，算 p-1 时预取 p
  float regA[2][Tm] = {0.0f};
  float regB[2][Tn] = {0.0f};

  int buffer_id = 0;  // ★ gemm9 double buffer：当前计算槽编号

  // ★ gemm2 block tile：block 负责的 tileC 左上角坐标
  int r0 = blockIdx.y * Bm;
  int c0 = blockIdx.x * Bn;

  int tid = threadIdx.x;

  // ★ gemm2 block tile：tileA 加载线程坐标
  // ★ gemm8 optimGEMM：tid % A_BLOCK_X → tid & (A_BLOCK_X - 1)
  int A_THREAD_Y = tid / A_BLOCK_X;
  int A_THREAD_X = tid & (A_BLOCK_X - 1);

  // ★ gemm2 block tile：tileB 加载线程坐标
  // ★ gemm8 optimGEMM：tid % B_BLOCK_X → tid & (B_BLOCK_X - 1)
  int B_THREAD_Y = tid / B_BLOCK_X;
  int B_THREAD_X = tid & (B_BLOCK_X - 1);

  // ★ gemm4 warp tile：计算 warp/lane 编号
  // ★ gemm8 optimGEMM：tid % WARP_SIZE → tid & (WARP_SIZE - 1)
  int warpId = tid / WARP_SIZE;
  int laneId = tid & (WARP_SIZE - 1);

  // ★ gemm4 warp tile：warp 在 block 中的方块坐标
  // ★ gemm8 optimGEMM：warpId % C_WARP_DIM_X → warpId & (C_WARP_DIM_X - 1)
  int warpX = warpId & (C_WARP_DIM_X - 1);
  int warpY = warpId / C_WARP_DIM_X;

  // ★ gemm4 warp tile：lane 在 warp 子块内的坐标（行优先线性映射）
  // int laneY = laneId / C_WARP_X;   // 0~3，沿 M 方向
  // int laneX = laneId % C_WARP_X;   // 0~7，沿 N 方向
  // 问题：相邻 laneId（i 和 i^1）的 laneX 不同 → 读 tileB 地址不同 → 无法触发广播 → 事务数 = 4
  //
  // ★ gemm7 z-order：z 形交错映射，让相邻线程（i 和 i^1）的 laneX 相同 → 触发广播 → 事务数降到 2
  // laneId 的 bit0 → laneY 低位，bit4 → laneY 高位，bit1~3 → laneX
  // 效果：bit0 翻转（i^1）只改 laneY，不改 laneX → 两线程读 tileB 同地址 → 广播成立
  int laneX = (laneId & 15) >> 1;
  int laneY = (laneId & 1) + ((laneId >> 4) << 1);

  // ★ gemm4 warp tile：当前线程在 blockC 中的坐标
  int C_THREAD_Y = warpY * C_WARP_Y + laneY;
  int C_THREAD_X = warpX * C_WARP_X + laneX;


  // ★ gemm9 double buffer：预先读取 k=0 的数据进 buffer[0]
#pragma unroll
  // ★ gemm8 optimGEMM：内层 j 循环已省略（A_BLOCK_X = Bk = 8，只有一次迭代）
  // ★ gemm8 optimGEMM：循环从 0 开始（原来从 A_THREAD_Y 开始导致 #pragma unroll 失效）
  for (int i = 0; i < Bm; i += A_BLOCK_Y) {
    int r = r0 + i + A_THREAD_Y;
    // ★ gemm5 向量化（转置）：As 按 [k][m] 存，外积时 A 的列变连续才能 float4 读
    // ★ gemm6 swizzling：^ 打乱 bank 落点，消除 bank conflict
    As[0][A_THREAD_X][(i + A_THREAD_Y) ^ (A_THREAD_X << 2)] =
        (r < M && A_THREAD_X < K) ? A[r * K + A_THREAD_X] : 0.0f;
  }

#pragma unroll
  for (int j = 0; j < Bm; j += B_BLOCK_X) {
    int c = c0 + j + B_THREAD_X;
    Bs[0][B_THREAD_Y][j + B_THREAD_X] = (B_THREAD_Y < K && c < N) ? B[B_THREAD_Y * N + c] : 0.0f;
  }

  __syncthreads();

  // K-Loop
  // ★ gemm2 block tile + gemm9 double buffer
  // 从 Bk 开始（k=0 已预取），超出 K 多一次收尾（为了完成最后一块的计算,所以要多跑一圈）
  for (int k = Bk; k < K + Bk; k += Bk) {

    // p-loop
    // ★ gemm9 double buffer，Bk+1 次迭代,p这里也同理,为了完成第p次搬运的计算 需要多跑一圈在p+1的时候做计算
    // p=0：只预取，不计算；p=1..Bk：计算 p-1 + 预取 p；p=Bk：只计算，不预取
#pragma unroll
    for (int p = 0; p < Bk + 1; ++p) {

      // 计算阶段：计算第 p-1 次循环的外积
      if (p > 0) {
#pragma unroll
        for (int i = 0; i < Tm; ++i) {
#pragma unroll
          for (int j = 0; j < Tn; ++j) {
            // ★ gemm3 thread tile：Tm×Tn 外积累加（核心）
            // ★ gemm8 optimGEMM：(p-1) % 2 → (p-1) & 1
            Ct[i][j] += regA[(p - 1) & 1][i] * regB[(p - 1) & 1][j];
          }
        }
      }

      // 预取阶段：从 shared 读取第 p 次循环需要的寄存器数据
      if (p < Bk) {
        // ★ gemm5 向量化读取：FLOAT4 一次读 4 个 float
        // ★ gemm3 thread tile：regA 缓存 A 列向量（外积）
        // ★ gemm8 optimGEMM：p % 2 → p & 1，i * 4 → i << 2
#pragma unroll
        for (int i = 0; i < (Tm >> 2); ++i) {
          int r = (C_THREAD_Y + i * C_BLOCK_Y) << 2;
          // ★ gemm6 swizzling：^ 还原 bank 打乱后的地址
          FLOAT4(regA[p & 1][i << 2]) = FLOAT4(As[buffer_id][p][r ^ (p << 2)]);
        }

        // ★ gemm5 向量化读取：FLOAT4 读 B 行向量
        // ★ gemm8 optimGEMM：p % 2 → p & 1，j * 4 → j << 2
#pragma unroll
        for (int j = 0; j < (Tn >> 2); ++j) {
          int c = (C_THREAD_X + j * C_BLOCK_X) << 2;
          FLOAT4(regB[p & 1][j << 2]) = FLOAT4(Bs[buffer_id][p][c]);
        }
      }
    }

    // ★ gemm9 double buffer：预取下一轮 tileA/tileB 进另一个缓冲槽（与上面计算重叠）
    if (k < K) {
      // ★ gemm5 向量化（转置存储）| ★ gemm6 swizzling
      // ★ gemm8 optimGEMM：内层 j 循环省略，循环从 0 开始
      int c = k + A_THREAD_X;
#pragma unroll
      for (int i = 0; i < Bm; i += A_BLOCK_Y) {
        int r = r0 + i + A_THREAD_Y;
        As[buffer_id ^ 1][A_THREAD_X][(i + A_THREAD_Y) ^ (A_THREAD_X << 2)] =
            (r < M && c < K) ? A[r * K + c] : 0.f;
      }

      int r = k + B_THREAD_Y;
#pragma unroll
      for (int j = 0; j < Bn; j += B_BLOCK_X) {
        c = c0 + j + B_THREAD_X;
        Bs[buffer_id ^ 1][B_THREAD_Y][j + B_THREAD_X] = (r < K && c < N) ? B[r * N + c] : 0.f;
      }

      __syncthreads();
    }

    buffer_id ^= 1;  // ★ gemm9 double buffer：切换缓冲槽（0→1 或 1→0）
  }

  // 将 Ct 写回全局内存 C
#pragma unroll
  for (int i = 0; i < Tm; ++i) {
    // ★ gemm8 optimGEMM：(i / 4) * 4 → (i >> 2) << 2，i % 4 → i & 3
    int r = r0 + (C_THREAD_Y << 2) + ((i >> 2) << 2) * C_BLOCK_Y + (i & 3);
#pragma unroll
    for (int j = 0; j < Tn; ++j) {
      // ★ gemm8 optimGEMM：同上
      int c = c0 + (C_THREAD_X << 2) + ((j >> 2) << 2) * C_BLOCK_X + (j & 3);
      if (r < M && c < N) { C[r * N + c] = Ct[i][j]; }
    }
  }
}

〇、模板参数怎么定？—— 性能的两个对立指标

这是整套 GEMM 优化的第一性原理。所有模板参数（Bm/Bn/Bk/blockSize/Tm/Tn）本质上都在调同两个量：计算访存比和 Occupancy。先理解它们为什么对立、为什么都不能走极端，再看具体参数怎么选才不会死记硬背。

0.1 核心结论：性能不是单指标最优，是两者乘积最高

一句话先记住：

算子性能 ≈ 计算访存比（能跑多快的上限）× Occupancy（能不能逼近上限）。两者互相拉扯，最优配置在乘积的峰值，不在任何单一指标的极值。

  • 计算访存比（arithmetic intensity） = 每从内存搬 1 字节，能做多少次 FLOP。它决定你是被算力卡住还是被带宽卡住 —— 即性能的天花板。
  • Occupancy（占有率） = 一个 SM 上实际驻留的 warp 数 / 硬件支持的最大 warp 数。它决定 SM 在等访存时有没有别的 warp 可切换来隐藏延迟 —— 即你能不能贴近天花板。

这两个量为什么对立？因为提高计算访存比的唯一手段是把 tile 做大（Bm/Bn/Tm/Tn 调大），而 tile 越大，每个 block / 每个线程占用的 shared memory 和寄存器越多，一个 SM 能塞下的 warp 就越少 → Occupancy 直接被砸下去。

0.2 为什么访存比太高反而不好？

很多人受"内存墙"影响，以为计算访存比越大越好 —— 错。超过某个点后纯属浪费，还白白赔掉 Occupancy。

关键概念是 Roofline 模型的脊点（ridge point）：

脊点=峰值算力 (FLOPS)峰值带宽 (Byte/s)\text{脊点} = \frac{\text{峰值算力 (FLOPS)}}{\text{峰值带宽 (Byte/s)}} 脊点=峰值带宽 (Byte/s)峰值算力 (FLOPS)​

  • 当你的计算访存比 < 脊点 → memory-bound，瓶颈是带宽，此时拉高访存比有实打实的收益。
  • 当你的计算访存比 > 脊点 → compute-bound，瓶颈已经变成计算单元，内存早就不是瓶颈了。再往上堆访存比，多出来的数据复用是纯浪费，而你为此付出的 tile 增大、Occupancy 暴跌、甚至寄存器溢出（register spilling）的代价却是实打实的。

💡 用 4060 Ti 的数字算一遍：fp32 算力 ≈ 22 TFLOPS，HBM 带宽 ≈ 288 GB/s → 脊点 ≈ 22000/288 ≈ 76。Bm=Bn=128 时计算访存比 = 32，还没到 76，理论上仍是 memory-bound。但 L2 cache 把有效带宽抬到约 716 GB/s（同行 block 复用 A、同列 block 复用 B，命中率轻松 >33%），有效脊点降到 22000/716 ≈ 31，于是 32 刚好越过。这才是 128 够用的真正原因 —— 不是访存比绝对值大，而是配合 L2 刚好越过脊点。越过之后再加大 tile，收益几乎为零，纯亏 Occupancy。

所以内存墙时代的正确心态不是"访存比越大越好"，而是"用最小的 tile 越过脊点" —— 越过脊点的最小 tile = 资源压力最小 = Occupancy 最高 = 延迟掩盖最好。

0.3 为什么 Occupancy 很高、访存比很低也不行？

反过来也是错的。如果你为了拉满 Occupancy 把 tile 做得很小（比如退化到 naive 的一线程一元素），计算访存比掉到 ≈1，远在脊点左侧，被死死焊在带宽上限上：

  • 此时 SM 上 warp 再多、Occupancy 再高（接近 100%），也只是一堆 warp 排队等同一条拥堵的内存总线。
  • 带宽是物理上限，多开 warp 并不会让 HBM 吐数据更快 → 计算单元大量空转 → 性能被带宽锁死。

💡 一句话对比两种极端：

  • 访存比太低（tile 太小）：Occupancy 拉满也没用，卡在带宽，算力闲置 → memory-bound 的死。
  • 访存比太高（tile 太大）：越过脊点后访存比是浪费，Occupancy 却被砸穿，warp 不够隐藏延迟 → 延迟暴露的死。

真正的最优：把计算访存比刚好提到脊点右侧一点点，剩下的资源全部留给 Occupancy。

0.4 高 Occupancy ≠ 高性能（进阶易错点）

还有一个反直觉但面试高频的点：即使不考虑访存比，Occupancy 本身也不是越高越好。

延迟隐藏有两条路，Occupancy 只是其中一条：

  1. 靠更多 warp（高 Occupancy）—— SM 等数据时切到别的 warp。
  2. 靠每个 warp 内更多独立指令（高 ILP，指令级并行）—— 一个 warp 内有多条互不依赖的指令可以连续发射，流水线照样满。

GEMM 恰恰大量依赖第二条：每个线程的 Tm×Tn = 64 个累加器全在寄存器里，提供了 64 条独立的 FMA 指令可以并行发射。这种高 ILP 让 GEMM 即使只有 30%~50% 的 Occupancy 也能喂饱计算单元 —— CUTLASS 的旗舰 kernel Occupancy 往往就只有 25%~50%。Vasily Volkov 的经典工作正是证明了「低 Occupancy + 高 ILP」可以打败「高 Occupancy」。

⚠️ 面试结论：不要把"调高 Occupancy"当成优化目标本身。Occupancy 只要"够用"（够隐藏延迟）就行，把寄存器留给 ILP 和数据复用往往更划算。能说出这一层，是 senior 和 junior 的分水岭。

第二节、Bm / Bn / Bk / blockSize 具体怎么选

理解了第零章的对立关系，再看具体参数就是"在这个权衡里找平衡点"，而非死记数字。

template<int Bm = 128, int Bn = 128, int Bk = 8, int blockSize = 256, ...>

2.1 Bm、Bn = 128：让计算访存比越过脊点

Bm、Bn 的核心作用是决定计算访存比。在每次 K-Loop 中，一个 block 完成 tileA（Bm×Bk）和 tileB（Bk×Bn）的矩阵乘，计算量 2·Bm·Bn·Bk FLOP；从 global memory 搬入的数据量 4·(Bm + Bn)·Bk 字节（忽略 shared 访存代价）。计算访存比：

2⋅Bm⋅Bn⋅Bk4⋅(Bm+Bn)⋅Bk=12(1Bm+1Bn)\frac{2 \cdot B_m \cdot B_n \cdot B_k}{4 \cdot (B_m + B_n) \cdot B_k} = \frac{1}{2\left(\frac{1}{B_m} + \frac{1}{B_n}\right)} 4⋅(Bm​+Bn​)⋅Bk​2⋅Bm​⋅Bn​⋅Bk​​=2(Bm​1​+Bn​1​)1​

两个关键推论：

  • 与 Bk 无关：Bk 同时出现在分子（多算 Bk 次）和分母（多搬 Bk 行），约掉了。Bk 不是用来调访存比的。
  • Bm = Bn 时最优：1/Bm + 1/Bn 在 Bm·Bn 固定（即 shared 预算固定）时，由均值不等式两者相等时和最小 → 访存比最大。所以方形 tile（128×128）在给定 shared 预算下访存比理论最优，瘦长 tile（如 256×64）同面积但访存比更差。

令 Bm = Bn = 128，计算访存比 = 32。结合第零章：4060 Ti 配合 L2 的有效脊点 ≈ 31，32 刚好越过 → 进入 compute-bound。Bm/Bn 过小会让访存比掉回脊点左侧（memory-bound）；过大则 shared 占用激增、Occupancy 暴跌，且访存比早已越过脊点纯属浪费。128 是"刚好越过脊点 + Occupancy 还保得住"的平衡点。

2.2 Bk = 8：被 float4 和 shared 占用夹出来的

Bk 不影响计算访存比，它被两个约束夹定：

  • 下界（向量化对齐）：每个线程搬运的 A 元素数 = Bm×Bk / blockSize = 128×8/256 = 4，正好凑一个 float4（128-bit，一条 LDG.128 指令）。Bk 太小（如 Bk=1）每次只搬 A 的一整列、沿 M 方向不连续，无法合并访存；且凑不满 float4。
  • 上界（Occupancy）：As[Bk][Bm] / Bs[Bk][Bn] 占的 shared 随 Bk 线性增长，Bk 太大 → 一个 SM 能驻留的 block 数减少 → Occupancy 下降。

取 8 在两者间平衡，还有个附带好处：A_BLOCK_X = Bk = 8 让加载循环只剩一次迭代，gemm8 里可以整个消掉（见第八章）。

2.3 blockSize = 256：Occupancy 的甜点

256 = 8 个 warp。这个数不是"为了 Occupancy 随便挑 256"，而是被 tile 形状算出来后、又恰好落在 Occupancy 甜点：

  • 被 tile 算出来：(Bm/Tm) × (Bn/Tn) = (128/8) × (128/8) = 16×16 = 256。定了 block tile（128×128）和 thread tile（8×8），256 是被迫的。
  • 恰好是 Occupancy 甜点：以 Ampere 一个 SM 上限 2048 线程算，2048/256 = 8 个 block/SM，给调度器足够多 warp 隐藏延迟；同时没撞到每 SM 最大 block 数上限。warp 太少（如 4 warp）延迟无法隐藏，warp 太多（如 16 warp）寄存器/shared 不够、可并发 block 反而减少。

2.4 这套参数是死的吗？—— 不是

这套模板是这一类 tiling 策略的参数化，具体数值随三件事变：

  • 硬件换代：A100→H100，SM 的寄存器/shared 容量、最大 block 数都变，甜点跟着移；Hopper 上可能用更大 tile + wgmma 异步指令，整套重来。
  • 矩阵形状：这套为大方阵优化。GEMV 或瘦长矩阵下，128×128 会让大量线程算空气，得换 split-K 或更扁的 tile。
  • 数据类型：fp16/bf16 走 Tensor Core，tile 必须对齐 Tensor Core 的 fragment 尺寸（如 16×16×16），且脊点剧烈右移（算力涨十几倍、带宽不变，脊点从 ~76 飙到 1000+）—— 同一 tile 在 fp32 是 compute-bound，换 fp16 可能瞬间变回 memory-bound，必须用更激进的 tile + 更深的 software pipeline 把访存比再拉上去。

所以真实的库（cuBLAS、CUTLASS）不是一套模板，而是几百套预编译 kernel + 启发式选择器，运行时按 M/N/K 和 GPU 型号挑一套。手写这个相当于 CUTLASS 里某一个特定 config 的展开。

⚠️ 方法论结论：真实调参不是套公式，是 profile 驱动 —— 用 Nsight Compute 看到底卡在 Occupancy、计算访存比、bank conflict 还是 register spilling，再针对性调对应参数。CUTLASS 直接暴力枚举上百组实测选优。

一、gemm1：naive GEMM

最朴素：一个线程负责 C 中一个元素。要算 M×N 个元素，就启动 M×N 个线程。每个线程跑一条 K 维内积循环。

__global__ void gemm_naive(float* A, float* B, float* C,
                           int M, int N, int K) {
  int row = blockIdx.y * blockDim.y + threadIdx.y;  // M 方向
  int col = blockIdx.x * blockDim.x + threadIdx.x;  // N 方向
  if (row < M && col < N) {
    float acc = 0.f;
    for (int k = 0; k < K; ++k)
      acc += A[row * K + k] * B[k * N + col];        // A 行 × B 列
    C[row * N + col] = acc;
  }
}

致命问题：访存全在 HBM，且毫无复用。 算一个 C 元素要读 K 个 A、K 个 B，共 2K 次 HBM 访问、2K 次 FLOP（1 乘 1 加），计算访存比 ≈ 1。而且相邻线程反复从 HBM 读同一行 A / 同一列 B —— 同样的数据被搬运无数次。这就是后面所有优化的出发点。

二、gemm2：block tile GEMM —— 最重要的第一刀

核心思想：从 block 的视角重新审视问题。 我们想把要用到的 A、B 数据搬进 shared memory 复用，但整块 A、B 太大放不进去 —— 于是切块（tile）+ K-loop。

2.1 为什么需要 K-loop？

shared memory 一个 block 只有几十 KB，放不下整行 A（K 列）或整列 B（K 行）。所以沿 K 维把它切成一段段 Bk 宽的小块，循环累加：每次只搬 Bm×Bk 的 A 块 和 Bk×Bn 的 B 块进 shared，算完局部贡献，再搬下一段。这就是 K-loop（外层循环 K 维）。

💡 K 是从哪来的？ K-loop 的 K 就是收缩维（contraction dim），即 A 的列 / B 的行。沿它分块、循环累加，是为了让 shared 装得下。

一个 block 负责 C 的一个 Bm×Bn 子块：
  for (k_tile : K / Bk):          ← K-loop
      load A[Bm,Bk] → shared
      load B[Bk,Bn] → shared
      __syncthreads()
      局部矩阵乘累加到 C_block
      __syncthreads()

2.2 block 内线程重排

默认 blockSize = 256。这 256 个线程要被重排成对应矩阵子块的形状去干活（比如 16×16），而不是一维排开。这一步是后面 thread tile / warp tile 的基础。

2.3 向量内积 vs 向量外积（K-loop 用的是外积）

这是 gemm2 里最反直觉、最值得讲透的点。同样算 C += A·B，有两种组织方式：

  • 向量内积（inner product）：固定 C 的一个元素，让 A 的一行 × B 的一列点乘出它。→ C 元素被读写很多次的视角，A、B 的元素也要被反复搬运。
  • 向量外积（outer product）：取 A 的一列（长 Bm）和 B 的一行（长 Bn），做外积得到一个 Bm×Bn 的矩阵，累加进 C_block。沿 K 把每个 (列,行) 外积累加起来。

💡 外积的关键优势（你卡住的点）：外积里 A 的一列、B 的一行只参与计算一次 —— 一次性读进来，外积算完即可丢弃，读下一段。每个 A/B 元素只被搬运一次；而内积里一个元素要被反复搬运。从结果看，外积省了访存次数。

更重要的是：外积天然把"被复用的数据"暴露成寄存器里的一小段向量（A 的一列 + B 的一行），这正是 gemm3/5/8 要榨取的对象 —— 用寄存器缓存、向量化读取、double buffer 全都建立在外积结构上。所以这里看不出优势很正常，它是为后面铺路的结构选择。

三、gemm3：thread tile GEMM —— 加 P-loop

在 block tile 基础上，让一个线程负责 C 的一个小块（多个元素），而不是一个元素。多出来的内层循环叫 P-loop。

💡 K 和 P 的命名来源（你问的点）：K-loop 的 K 是收缩维（A 列 / B 行），是跨 block 共享的外层累加。P-loop 的 P 通常指 partial / per-thread 的内层循环 —— 在一个 Bk 段内，遍历这个线程负责的那几个 (m,n) 局部坐标，做外积累加。（注：命名是原博客作者的习惯，不是 CUDA 官方术语，理解成"K=外层 contraction 循环，P=线程内的局部累加循环"即可。）

为什么这样能拉高计算访存比：一个线程算 Tm×Tn 个 C 元素时，从 shared 读进寄存器的 A 的 Tm 个数 + B 的 Tn 个数，可以做 Tm×Tn 次乘加。访存 Tm+Tn，计算 Tm×Tn，比值从 naive 的 ~1 拉到 Tm·Tn/(Tm+Tn)。这就是 thread tile 的全部意义：寄存器复用。

四、gemm4：warp tile GEMM —— 重排 warp 的 C 分块形状

核心：在 warp 这一层，调整一个 warp 负责的 C 子块形状，让计算访存比更高。

优化的位置：shared memory → register 这一步。数据已经在 shared 里了，warp 的分块形状决定了每次做外积时要从 shared 广播多少行 A、多少列 B —— 形状越接近正方形，shared 读次数越少，计算访存比越高。

正常排法是图7那种（warp 横向铺开，C_block 16×16，warp0 占 16×2 等），改成图8那种（warp 占 8×4 的方块形）后，计算访存比更高。

💡 为什么方块形访存比更高（你卡住的逻辑）：一个 warp（32 线程）做外积时，要从 shared 读进 A 的一段（行数 = warp 覆盖的 M 高度）和 B 的一段（列数 = warp 覆盖的 N 宽度）。

  • 若 warp 是 16×2 长条：读 16 个 A + 2 个 B = 18 次访存，算 16×2=32 次 → 比值 32/18 ≈ 1.8。
  • 若 warp 是 8×4 方块：读 8 个 A + 4 个 B = 12 次访存，算 8×4=32 次 → 比值 32/12 ≈ 2.7。

同样的计算量（32），方块形周长最小 → 访存最少 → 计算访存比最高。 这就是"为什么倾向正方形分块"的本质：计算量 ∝ 面积，访存量 ∝ 周长，正方形周长/面积比最优。 因内存墙，我们永远想要更高的计算访存比（= 更高 arithmetic intensity），让本就高算力低带宽的 GPU 跑在算力上限而非带宽上限。

五、gemm5：向量化读取 —— 转置 tileA + float4

核心：用 float4（128-bit）一次读 4 个连续 float，把访存指令数砍到 1/4。

5.0 float4 为什么只能加在 shared → register，而不能加在 HBM → shared？

核心：float4 要求起始地址能被 4 整除（16B 对齐）。

  • shared → register：tileA/tileB 的列数是我们自己定的（Bk、Bn 都取 2 的幂），每行 index 独立从 0 开始计，只跟这一小块矩阵的列数有关 → 完全可控，天然对齐。
  • HBM → shared：HBM 把整个矩阵压成一维，线性地址 = row * N + col。每行的起始地址不是从 0 重算，而是受全局列数 N 累积影响。并且，N 是用户传入的，是不可控的，一旦 N 不是 4 的倍数，每行起始地址就不对齐，float4 报错。

结论：HBM 是全局一维地址，行起始受用户给的 N 控制、不可控；shared/register 是局部地址，行 index 独立可控。float4 只能安全地用在后者。

5.1 为什么要先转置 tileA？

向量化读取要求被读的 4 个元素在内存里连续。外积里要取 A 的一列（沿 M 方向），但 A 在 shared 里是行主序存的，一列的元素并不连续 → 没法 float4。先把 tileA 转置存进 shared，“列"就变成了"行”、变连续了，才能向量化读。

5.2 向量化前后对比

向量化之前（gemm4）：每次外积循环，每个线程从 tileA/tileB 逐个读 float，共需读 Bk 次 A + Bk 次 B，访存指令数 = 2 * Bk。

向量化之后（gemm5）：改用 float4 一次读 4 个，访存指令数直接砍到 2 * Bk / 4，减少 4 倍。

六、gemm6：swizzling 解决 bank conflict

引入 shared memory 后，按列访问 tile 会撞 bank（同列元素落同一 bank → 串行）。用 swizzling（异或打乱列在 bank 里的落点）解决，零显存浪费。

原理同 transpose 笔记的 swizzle：更换每一列的存储位置，使得虽然按列读取，但因存储位置不同而落在不同 bank，从而合并为一次内存事务。

6.1 为什么是 x << 2？（异或项的两个旋钮）

As[0][A_THREAD_X][(i + A_THREAD_Y) ^ (A_THREAD_X << 2)] = ...;

这个异或表达式藏着两个独立的设计决定：移几位、绑哪一维。分开看。

旋钮一：移 2 位 —— 为了不打散 float4

我们用 float4 一次读 4 个连续元素，所以 swizzle 必须 4 个一组地打乱、组内 4 个保持连续。二进制上，4 个连续坐标只差最低 2 位（…00 / …01 / …10 / …11），只要异或不碰最低 2 位，这 4 个就永远连在一起。

<< 2 做的就是这件事：把异或的扰动全部推到第 3 位以上，最低 2 位原封不动 → 每个 float4 组内连续性自动保住。

连续坐标:  ...0 00 | ...0 01 | ...0 10 | ...0 11    ← 只差最低 2 位
异或项 << 2: x x x 0 0                              ← 扰动全在高位，碰不到最低 2 位
结果:      4 个一组被整体搬到新 bank，但组内仍连续 ✓

所以移几位这个旋钮只跟向量化宽度挂钩：float4 = 4 = 2² → 移 2 位。跟 bank 撞不撞无关。

旋钮二：绑 x —— 因为要保连续的是 y（转置翻转了连续维）

定下"最低 2 位不许异或"后，问题变成：最低 2 位属于哪一维？必须保哪一维的连续性？

关键在于 As 存的是转置后的 tileA（As[Bk][Bm]，第二维 k、第三维 m）。外积要按 A 的列 取数，转置后列变成了第三维（m 方向），float4 正是沿这一维连续读 —— 也就是说：

转置把"连续维"从 x 翻成了 y。 要保护连续性的是 i + A_THREAD_Y 这一维（y/m 方向），它的最低 2 位绝不能被异或碰。

异或项既然要避开 y，就只能由另一维 x 来驱动 → 所以是 A_THREAD_X << 2。这样 y 那一维 4 个 4 个连续不被打乱，而不同的 x 给每行配一个不同的扰动量，把 bank 落点错开。

(i + A_THREAD_Y) ^ (A_THREAD_X << 2)
 └──────┬───────┘    └──────┬──────┘
 连续维（转置后的 y/m）      用另一维 x 驱动异或
 float4 靠它连续           └ <<2 把扰动推到高位，
 → 异或绝不能碰它             不污染连续维的低 2 位

💡 一句话判定法（不用背结论）：先问"这一拍 float4 的 4 个连续元素，下标靠哪一维在低位变？"——那一维就是连续维，异或必须避开它，绑到另一维。这里因为做了转置，连续维是 y，所以绑 x。换个没转置的场景，结论可能反过来。

🔗 bank conflict 只是这一切的底层落点：异或最终把"按列访问会撞同一 bank"的那些地址散到不同 bank。但主线是 “为 float4 服务的错位”，bank 是结果不是出发点。

七、gemm7：z-order GEMM —— 改善 float4 访存合并

7.1 float4 之后访存合并逻辑变了

普通 float32（每次读 4B），half-warp 16 个线程共读 16×4B = 64B，一次 128B 事务可以装两个 half-warp，32 个 bank 各不同就能合并，条件宽松。

向量化之后每次读 float4（16B），half-warp 16 线程共读 16×16B = 256B，远超 128B 的单次事务上限。合并的关键不再是"落在不同 bank"，而是要触发广播机制：

触发广播机制（两个 half-warp 的事务合并成一个）需满足以下条件之一：

  • 对 warp 内所有活跃的第 i 号线程，第 i xor 1 号线程不活跃或访存地址与其一致；
  • 对 warp 内所有活跃的第 i 号线程，第 i xor 2 号线程不活跃或访存地址与其一致。

满足其中一个条件，两个 half-warp 的 memory transaction 才能合并成一个。

7.2 z-order 解决什么

z-order 前（行优先映射）：quarter-warp（8 线程）内每个线程负责 C 的不同行，从 tileB shared 里读的是同一行的不同列位置 —— 地址各不相同，i xor 1 条件不满足，无法触发广播，事务数 = 4。

z-order 后（z 形交错映射）：相邻两个线程（i 和 i xor 1）被映射到读取 tileB 的同一地址，满足广播机制第一条，两个 half-warp 的事务合并，事务数从 4 降到 2，访存量减半。

7.3 代码对比：gemm4 行优先 → gemm7 z 形映射

gemm4 的 lane 映射（行优先，线性排开）：

// warp 内 laneId 0~31，按行优先直接铺到 C_WARP_Y × C_WARP_X 的子块
// C_WARP_Y = 4（M 方向），C_WARP_X = 8（N 方向）
int laneY = laneId / C_WARP_X;   // 0~3，沿 M 方向
int laneX = laneId % C_WARP_X;   // 0~7，沿 N 方向

laneId 0~7 在同一行（laneY=0），读 tileB 时每人负责不同列：lane0 读 col0，lane1 读 col1……相邻的 i 和 i xor 1 地址不一致 → 无法触发广播 → 事务数 = 4。

gemm7 的 z 形映射：

// z 形交错：把 laneId 的 bit[0] 分给 Y，bit[4] 分给 Y，中间 bit[1~3] 给 X
int laneY = (laneId & 1) + ((laneId >> 4) << 1);   // 取 bit0 和 bit4，组成 0~3
int laneX = (laneId & 15) >> 1;                     // 取 bit1~3，组成 0~7

拆开看每个 bit 的去向：

laneId bit 4 3 2 1 0
分配给 laneY[1] laneX[2] laneX[1] laneX[0] laneY[0]

效果：laneId 相邻的两个线程（i 和 i xor 1，即 bit0 翻转）→ laneY 不同、laneX 相同。读 tileB 时（只用 laneX 决定列），地址完全相同 → 触发广播机制第一条 → 事务从 4 降到 2。

为什么是 (laneId & 1) + ((laneId >> 4) << 1)？

  • laneId & 1：取 bit0，得 0 或 1，这是 laneY 的低位
  • (laneId >> 4) << 1：取 bit4（值 0 或 1），左移 1 位变成 0 或 2，这是 laneY 的高位
  • 相加后 laneY ∈ {0,1,2,3}，正好覆盖 C_WARP_Y=4 的范围

为什么是 (laneId & 15) >> 1？

  • laneId & 15：保留 bit0~3（屏蔽 bit4）
  • >> 1：右移 1 位，扔掉 bit0（bit0 已经给了 laneY），剩下 bit1~3，范围 0~7
  • 正好覆盖 C_WARP_X=8 的范围

八、gemm8：optimGEMM —— 位运算 + 循环优化

这一步是在 gemm7 基础上做指令级优化，不改变算法结构，只消除低效运算。完成后计算时间缩短至 1.85 ms，相对效率达到 77.8%，提高约 19%。

8.1 优化一：除法/取余 → 位运算

在程序中，除法运算和求余运算都是一个复杂的计算过程，通常需要多个时钟周期才能完成。而由于计算机的二进制特性，当对元素进行 2 的幂次方形式的除法或取余运算时，可以转换为更高效的位运算写法：

  • x / 2^n 可以转换为 x >> n（同理，2 的幂次乘法可以写成左移位运算形式）
  • x % 2^n 可以转换为 x & (2^n - 1)

现有 GEMM 算子的代码中，几乎所有的乘法、除法和取余运算都是与 2 的幂次方形式进行的，因此可以全部以位运算形式代替。例如：

// before（gemm7）
int A_THREAD_X = tid % A_BLOCK_X;          // A_BLOCK_X = 8 = 2^3
int laneId     = tid % WARP_SIZE;           // WARP_SIZE = 32 = 2^5
int warpX      = warpId % C_WARP_DIM_X;    // C_WARP_DIM_X = 2 = 2^1

// after（gemm8）
int A_THREAD_X = tid & (A_BLOCK_X - 1);    // tid & 7
int laneId     = tid & (WARP_SIZE - 1);    // tid & 31
int warpX      = warpId & (C_WARP_DIM_X - 1); // warpId & 1

8.2 优化二：循环优化

tileA 加载循环原来的写法（gemm7）存在两个问题：

问题 a：内层循环可以省略。 在参数设置时，A_BLOCK_X 和 Bk 全部取值为 8，因此外层 j 循环不有且只有一次迭代（j = A_THREAD_X），整个 j 循环可以直接删掉：

// before：多余的外层 j 循环
for (int j = 0; j < Bk; j += A_BLOCK_X) {   // A_BLOCK_X = Bk = 8，只循环一次
    for (int i = A_THREAD_Y; i < Bm; i += A_BLOCK_Y) {
        As[A_THREAD_X][i ^ (A_THREAD_X << 2)] = A[...];
    }
}

// after：去掉 j 循环，j = A_THREAD_X 直接用
for (int i = 0; i < Bm; i += A_BLOCK_Y) {
    int r = r0 + i + A_THREAD_Y;
    As[A_THREAD_X][(i + A_THREAD_Y) ^ (A_THREAD_X << 2)] = A[...];
}

问题 b：#pragma unroll 没有起到作用。 #pragma unroll 的作用是展开已知次数的循环运算，而原代码中 i 的初始值为 A_THREAD_Y，这是一个在不同 thread 中数值不同的变量，编译器无法预先确定循环次数，因此 #pragma unroll 无效。

修复方法：将循环改为从 0 开始，把 A_THREAD_Y 移进循环体内作为偏移量。这样循环次数 Bm / A_BLOCK_Y 在编译时可知，#pragma unroll 才能真正展开：

// before：i 从 A_THREAD_Y 开始，unroll 无效
#pragma unroll   // ← 实际上没有展开！
for (int i = A_THREAD_Y; i < Bm; i += A_BLOCK_Y) { ... }

// after：i 从 0 开始，循环次数 = Bm/A_BLOCK_Y 编译期可知，unroll 生效
#pragma unroll   // ← 真正展开 Bm/A_BLOCK_Y 次
for (int i = 0; i < Bm; i += A_BLOCK_Y) {
    int r = r0 + i + A_THREAD_Y;   // A_THREAD_Y 作为偏移放进体内
    ...
}

九、gemm9：double buffer —— 串行变并行（最终背诵版）

这是面试要手撕的最终版本，必须理解到能默写。核心：把"load 数据"和"compute"从串行改成并行流水。

9.1 理论分析

问题：串行的 load → compute 浪费

朴素流水（图14）：每个阶段 p 先 load regA/regB，再 FFMA（fused multiply-add）。load 和 compute 串行，计算单元在等数据时空转。

解法：double buffer 预取，load 与 compute 重叠

double buffer（双缓存）：开两份寄存器/shared，当前在算 buffer[0] 的同时，预取下一段进 buffer[1]，下一拍交换。如图15，p=1 的 load 和 p=0 的 FFMA 在时间上重叠了 —— 访存延迟被计算掩盖，流水线打满。

💡 本质（你说的"串行变并行"）：不是真的多开线程并行，而是指令级并行（ILP）+ 延迟隐藏 —— 让访存指令和计算指令在硬件流水里 overlap，使 GPU 在等 HBM/shared 数据时不空转，持续做 FFMA。这是把计算访存比的优化"落地"到时间维度的最后一步。

双缓冲的两个层次

double buffer 同时作用在两个层次，相辅相成：

层次 缓冲对象 目的
shared memory 级 As[2][Bk][Bm] / Bs[2][Bk][Bn] K-loop 的下一个 tile 从 HBM 预取时，当前 tile 的计算不被打断
register 级 regA[2][Tm] / regB[2][Tn] p-loop 的下一个外积从 shared 预取时，当前外积计算不被打断

两级流水叠加，几乎完全掩盖 HBM→shared 和 shared→register 两段访存延迟。

9.2 易错点

易错点1：k-loop 和 p-loop 的循环次数要 +1

k-loop 从 Bk 开始、上限是 K + Bk（而不是 K）；p-loop 从 0 到 Bk + 1（而不是 Bk）。两者都要多跑一圈，原因完全对称：

  • k-loop 多一圈：k=0 的 tile 在循环外已经预读进 buffer[0]，k-loop 从 Bk 开始跳过了 k=0。但最后一段 tile（k = K - Bk 那一次预读）的计算还没完成，必须多跑一圈来收尾这最后一块的 FFMA。
  • p-loop 多一圈：同理，p=0 只预取、不计算；p=1 开始才算 p-1 的外积。最后一次从 shared 预取的寄存器数据（第 Bk-1 次），要等 p = Bk 那圈才能被算完。
// k-loop：从 Bk 开始，上限 K + Bk（多一圈收尾最后一块的计算）
for (int k = Bk; k < K + Bk; k += Bk) {
    // p-loop：0 到 Bk（共 Bk+1 次，多一圈收尾最后一次预取的计算）
    for (int p = 0; p < Bk + 1; ++p) {
        ...
    }
}

直觉记法：double buffer 是"先读后算，错位一拍"——有多少次"读"，就得有多少次"算"来收尾；读写各差一圈，所以各 +1。

易错点2：k-loop 的预读取必须写在循环外面

k-loop 里的 if (k < K) 预读 HBM→shared 这段代码，必须放在循环体外、p-loop 之后，而不是打包进 p-loop 内部。

原因：k-loop 不能用 #pragma unroll 展开，因为 K 是运行时才知道的参数（用户调用算子时传入矩阵大小），编译器看不到 K 的值，无法在编译期确定循环次数，unroll 根本不生效。

如果把预读写进 p-loop 内部，代码逻辑上也能跑，但你需要在 p-loop 里加额外的 if 判断（比如"p 等于某个值才触发 HBM 预读"），不仅让 p-loop 逻辑变复杂，还因为 k-loop 无法 unroll 而多出运行时分支代价。

正确写法：预读独立放在 p-loop 结束后、k-loop 末尾：

for (int k = Bk; k < K + Bk; k += Bk) {
    // p-loop（shared→register 预取 + FFMA，可以 unroll 因为 Bk 编译期已知）
    #pragma unroll
    for (int p = 0; p < Bk + 1; ++p) { ... }

    // HBM→shared 预读：独立在 p-loop 外，k-loop 无法 unroll，放外面最干净
    if (k < K) {
        // load As[buffer_id ^ 1], Bs[buffer_id ^ 1] from HBM
        ...
        __syncthreads();
    }

    buffer_id ^= 1;
}

对比：p-loop 的 Bk 是模板参数，编译期已知，所以 p-loop 可以 unroll，把预取和计算的 overlap 完全展开。k-loop 的 K 运行时才知道，unroll 无效，所以 HBM 预读不必也不能往里塞。

十、易错点速查 List

这是咱们讨论过程中你提到的卡点 + 我补充的易错点，持续更新。

# 易错点 正确理解
1 GEMM 优化的主线是什么？ 逼近计算访存比 × Occupancy 的乘积峰值。前者决定性能天花板（compute-bound 还是 memory-bound），后者决定能否贴近天花板。两者对立，最优在乘积峰值而非单指标极值。
2 K-loop 的 K 从哪来？ K = 收缩维（A 列 / B 行）。沿它分块循环累加，是为了让 tile 装进 shared。
3 P-loop 的 P 从哪来？ 非官方术语，理解为 per-thread / partial 的线程内局部累加循环（在一个 Bk 段内遍历该线程负责的 (m,n)）。
4 内积和外积代码看起来一样，有区别吗？ 编译器优化后性能差不多，但外积的结构关键：A 列 / B 行只读一次即丢弃，每元素只搬一次；且把可复用数据暴露成寄存器向量，为后续优化铺路。
5 外积省了什么？ 省访存次数。内积里一个元素被反复搬运，外积里只搬一次。
6 为什么 warp 倾向方块形分块？ 计算量 ∝ 面积，访存量 ∝ 周长。同面积下正方形周长最小 → 访存最少 → 计算访存比最高。
7 向量化读取为什么要先转置 tileA？ float4 要求 4 元素连续。外积取 A 的列不连续，转置后列变行、变连续，才能向量化。
8 为什么 GEMM 比 reduce 更适合向量化？ 待深入讨论。初步：GEMM 访存是规则稠密 tile、天然对齐；reduce 受 strided/树形结构限制，瓶颈在带宽利用而非指令数。
9 swizzling 解决什么？和 padding 啥关系？ 解决 shared memory 的 bank conflict。异或打乱列的 bank 落点，零显存浪费（vs padding 费显存）。
9.1 swizzle 为什么是 x << 2，不是 y << 2？ 两个独立旋钮。移 2 位：float4=2² → 把异或扰动推到高位、保最低 2 位不变，使每组 4 个连续元素不被打散（与 bank 无关）。绑 x：要保连续性的是 float4 的连续维，而 As 存转置后的 tileA，转置把连续维从 x 翻成了 y，所以异或必须避开 y、绑到 x。判定法：问"float4 的 4 连续元素靠哪维低位变"→ 那维是连续维、异或避开它、绑另一维。
9.2 "只要保最后两位不变，谁左移都一样"对吗？ 错。保最后两位（移 2 位）只保证 float4 组内连续，是旋钮一；绑哪一维是旋钮二，由"谁是连续维"决定，唯一解。两个旋钮正交，别混。
10 z-order 优化点在哪？ float4 后每次读 16B，half-warp 共 256B 超 128B 事务上限，合并必须靠广播机制（需 i xor 1 或 i xor 2 地址一致）。行优先映射下相邻线程读 tileB 不同列、地址各异，不触发广播，事务数=4；z-order 让相邻线程对齐到同一地址，触发广播，事务数降到 2。
11 gemm8 的 #pragma unroll 为什么之前没起作用？ 原循环从 A_THREAD_Y 开始，该值在不同 thread 中不同，编译器无法确定循环次数，unroll 失效。改为从 0 开始、A_THREAD_Y 作为体内偏移后，次数 = Bm/A_BLOCK_Y 编译期可知，才能真正展开。
12 double buffer 的"串行变并行"是多线程吗？ 不是。是指令级并行 + 延迟隐藏，让 load 和 compute 在硬件流水里 overlap，等数据时不空转。
13 double buffer 作用在几个层次？ 两个：shared memory 级（掩盖 HBM→shared 延迟）+ register 级（掩盖 shared→register 延迟），两级流水叠加。
14 计算访存比是越大越好吗？ 不是。超过 Roofline 脊点后就是 compute-bound，内存不再是瓶颈，多出的复用纯浪费，还白赔 tile 增大带来的 Occupancy 暴跌。正确目标是"用最小 tile 刚好越过脊点"。
15 Occupancy 是越高越好吗？ 不是。① 访存比太低时，Occupancy 拉满也卡在带宽。② 延迟隐藏还能靠 ILP，GEMM 每线程 64 个寄存器累加器提供高 ILP，30%~50% Occupancy 就够喂饱 ALU。CUTLASS 旗舰 kernel Occupancy 常只有 25%~50%。
16 脊点（ridge point）怎么算？ 脊点 = 峰值算力 / 峰值带宽。算访比 < 脊点 → memory-bound；> 脊点 → compute-bound。4060 Ti fp32 脊点 ≈ 76，配合 L2 有效带宽后降到 ≈ 31，所以算访比 32 刚好越过。
17 blockSize=256 是先定的还是推出来的？ 推出来的。先定 block tile（128×128）和 thread tile（8×8），256 = (128/8)×(128/8) 被迫得出，且恰好落在 2048/256=8 block/SM 的 Occupancy 甜点。
18 为什么 Bm=Bn（方形 tile）最优？ 访存比 = 1 / [2(1/Bm+1/Bn)]。shared 预算固定即 Bm·Bn 固定时，由均值不等式 Bm=Bn 使分母最小、访存比最大。瘦长 tile 同面积但访存比更差。
19 为什么 Bk=8？ 被 float4 和 shared 占用夹定。下界：每线程搬 Bm·Bk/blockSize=4 个 float 凑满 float4；上界：Bk 太大 shared 占用涨、Occupancy 掉。与计算访存比无关（Bk 在分子分母约掉）。
20 换数据类型/硬件后参数还一样吗？ 不一样。fp16 走 Tensor Core 脊点从 ~76 飙到 1000+，同 tile 可能从 compute-bound 变回 memory-bound，需更激进 tile + 更深 pipeline。真实库（cuBLAS/CUTLASS）是几百套预编译 kernel + 启发式选择器，profile 驱动而非套公式。
21 double buffer 里 k-loop 和 p-loop 循环次数为什么要 +1？ 两者都是"先读后算、错位一拍"的收尾问题。k-loop 从 Bk 开始（k=0 已在循环外预读），上限必须是 K + Bk 才能算完最后一个 tile；p-loop 从 0 到 Bk + 1（共 Bk+1 次），p=0 只预取不算，最后一圈 p=Bk 只算不预取，用来收尾第 Bk-1 次预取的寄存器数据。
22 k-loop 的 HBM 预读为什么必须写在 p-loop 外面，不能放进 p-loop 里？ k-loop 无法 unroll：K 是运行时参数，编译器不知道循环次数，#pragma unroll 对 k-loop 无效。把预读放进 p-loop 需要额外的运行时 if 判断，逻辑更复杂；放外面代码最干净，也不影响性能。反之，p-loop 的 Bk 是编译期已知的模板参数，可以 unroll，两个循环性质不同。
