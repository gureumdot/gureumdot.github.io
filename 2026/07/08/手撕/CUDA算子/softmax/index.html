Softmax

推理优化 CUDA 算子优化
手撕 CUDA算子
Publish Date:   2026-07-08

手撕目标

Softmax 对一行做 y_i = exp(x_i) / Σ_j exp(x_j)。它本质是两次 reduce（一次 max、一次 Σexp）夹一次逐元素归一化，是 attention、LM head、MoE gating 里最高频的算子。

两章：

  • CPU 版：从 safe softmax（3 pass）到 online softmax（2 pass），把"减 max 防溢出""running max 边扫边 rescale"讲清楚——这套 rescale 递推就是 FlashAttention 的地基。
  • GPU 版：一个 block 处理一行，用 warp/block reduce 求 max 和 Σexp，再逐元素写回。

一、CPU 版本

1. safe softmax（3 pass baseline）

直接算 exp(x_i) 会溢出（x 稍大 expf 就 inf）。减去行 max 后每个指数 ≤ 0，永不溢出，且结果不变（分子分母同乘 exp(-max) 约掉）：

void softmax_safe(float* x, float* y, int N) {
    // pass 1: 求 max
    float m = -INFINITY;
    for (int i = 0; i < N; i++) m = std::max(m, x[i]);

    // pass 2: 求 Σexp(x - max)
    float d = 0;
    for (int i = 0; i < N; i++) d += std::expf(x[i] - m);

    // pass 3: 归一化
    for (int i = 0; i < N; i++) y[i] = std::expf(x[i] - m) / d;
}

三趟里 pass 1、pass 2 都要把整行 x 从内存读一遍。online softmax 的目标就是把这两趟合成一趟，少读一遍 x。

2. online softmax（2 pass）

void softmax_online(float* x, float* y, int N) {
    // pass 1: 一次遍历同时求 max 和 sum
    float m = -INFINITY;   // running max
    float d = 0;           // running sum (denominator)
    for (int i = 0; i < N; i++) {
        float m_prev = m;
        m = std::max(m, x[i]);
        d = d * std::expf(m_prev - m) + std::expf(x[i] - m);
    }

    // pass 2: 归一化
    for (int i = 0; i < N; i++) {
        y[i] = std::expf(x[i] - m) / d;
    }
}

核心是那行 d = d * expf(m_prev - m) + expf(x_i - m)。 为什么能一趟同时维护 max 和 sum：

  • d 始终定义为"相对当前 running max m 的部分和"：d = Σ_{j≤i} exp(x_j - m)。
  • 来了新元素 x_i，running max 从 m_prev 更新到 m = max(m_prev, x_i)。此前累加的 d_prev 是按旧基准 m_prev 算的，基准一变，每一项都要换算：exp(x_j - m) = exp(x_j - m_prev) · exp(m_prev - m)，所以整块 d_prev 统一乘上 rescale 因子 exp(m_prev - m)。
  • 换算完再把新项 exp(x_i - m) 加进去。
  • 若 m 没变（x_i 不是新的 max），m_prev - m = 0，rescale 因子 = 1，退化成普通累加，不影响正确性。

一趟扫完，m 就是全局 max，d 就是 Σ exp(x - m)，和 3-pass 版数值完全一致。代价：pass 1 里每步多一次 expf(m_prev - m)（换算旧和），换来少读一整趟 x。当 x 装不进 cache 时，省下的访存远大于多算的这点 exp——这正是 FlashAttention 把 online softmax 搬进 attention、边算 QKᵀ 边 rescale 累加、从不物化整行的动机。

pass 2 仍要再扫一趟 x 做归一化，所以叫 2-pass。真正做到 1-pass 的是 FlashAttention：它在算 O 的同时把 rescale 施加到已累加的输出上，连归一化那趟也省了。

二、GPU 版本

并行策略：一个 block 处理一行（softmax 沿最后一维 N，输入 M×N）。block 内 blockSize 个线程用 grid-stride 扫这一行，两次 block reduce 分别求出行 max 和 Σexp，广播给所有线程后各自归一化写回。

reduce 要用 shfl_xor（蝶式）而非 shfl_down（漏斗）：softmax 里每个线程都要拿到 max / sum 给自己那几个元素用，shfl_xor 让 warp 内每个 lane 都拿到归约结果，而 shfl_down 结果只落在 lane 0（详见 reduce）。

手撕代码

#define WARP_SIZE 32

// shfl_xor：每个 lane 都拿到结果（softmax 全员要用 max/sum）
template <int kWarpSize = WARP_SIZE>
__device__ __forceinline__ float warpReduceMax(float val) {
#pragma unroll
    for (int delta = kWarpSize >> 1; delta >= 1; delta >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, delta));
    return val;
}

template <int kWarpSize = WARP_SIZE>
__device__ __forceinline__ float warpReduceSum(float val) {
#pragma unroll
    for (int delta = kWarpSize >> 1; delta >= 1; delta >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, delta);
    return val;
}

// block reduce：结果广播到 block 内所有线程（每个线程读回全部 warp 部分值再归约）
template <int blockSize, bool IsMax>
__device__ __forceinline__ float blockReduce(float val) {
    constexpr int NUM_WARPS = (blockSize + WARP_SIZE - 1) / WARP_SIZE;
    __shared__ float smem[NUM_WARPS];
    int lane = threadIdx.x & (WARP_SIZE - 1);
    int wid  = threadIdx.x >> 5;

    val = IsMax ? warpReduceMax(val) : warpReduceSum(val);   // warp 内先归约
    if (lane == 0) smem[wid] = val;                          // 每个 warp 写一个部分值
    __syncthreads();

    // 每个线程读回全部 NUM_WARPS 个部分值，自己再归约一次 → 全员拿到 block 结果，无需二次 sync
    float res = IsMax ? -INFINITY : 0.f;
#pragma unroll
    for (int i = 0; i < NUM_WARPS; i++)
        res = IsMax ? fmaxf(res, smem[i]) : res + smem[i];
    return res;
}

// 一个 block 处理一行 x[row][0..N)
template <int blockSize>
__global__ void softmax_kernel(const float* x, float* y, int N) {
    const float* xrow = x + (size_t)blockIdx.x * N;
    float*       yrow = y + (size_t)blockIdx.x * N;
    int tid = threadIdx.x;

    // pass 1: 行 max（grid-stride 线程内累加 → block reduce）
    float m = -INFINITY;
    for (int i = tid; i < N; i += blockSize) m = fmaxf(m, xrow[i]);
    m = blockReduce<blockSize, true>(m);

    // pass 2: Σexp(x - max)
    float d = 0.f;
    for (int i = tid; i < N; i += blockSize) d += __expf(xrow[i] - m);
    d = blockReduce<blockSize, false>(d);

    // pass 3: 归一化写回
    float inv = 1.f / d;
    for (int i = tid; i < N; i += blockSize) yrow[i] = __expf(xrow[i] - m) * inv;
}

调用配置：

constexpr int blockSize = 256;
softmax_kernel<blockSize><<<M, blockSize>>>(d_x, d_y, N);
//             ├模板参数┤   ├──launch 配置──┤ ├─函数实参─┤
//                          grid = M 行，每 block 一行；block = blockSize 线程

逐条拆解

一 block 一行 + grid-stride

grid = M，每个 block 独吞一行，block 之间天然无依赖、不用跨 block 通信。行长 N 可能远大于 blockSize，所以每个线程用 for (i = tid; i < N; i += blockSize) grid-stride 吃多个元素，寄存器里先做线程内局部归约，再进 block reduce——和 reduce 的 grid-stride 同一套路。

为什么 3 pass 而不 online

GPU 版这里写的是 safe softmax（3 pass）：block reduce 求 max、block reduce 求 Σexp、逐元素归一化。CPU 章的 online 递推在单线程串行下才直接成立（m_prev → m 有明确先后）；并行归约里各线程/各 warp 的局部 max、局部 sum 是同时算的，要合并两个 (m, d) 分块得用带 rescale 的二元合并算子：

合并 (m1,d1) 与 (m2,d2)：  m = max(m1,m2)，  d = d1·exp(m1-m) + d2·exp(m2-m)

这个算子满足结合律，可以塞进 warp/block reduce 做成真正的并行 online softmax（省掉一趟对 x 的读取）。但写起来要 reduce 一个 float2 (m,d)、每步做 rescale，代码更绕；行能放进 shared/L2 时收益也有限。面试先手撕清楚 3-pass 版，再口述"online 版把 max 和 sum 融进一次 reduce、合并时按上式 rescale"，就是 FlashAttention 的核心。

__expf vs expf

kernel 里用 __expf（intrinsic，走 SFU 硬件单元，快、精度略低），推理 softmax 足够；要高精度用 expf。

三、易错点速查 List

# 易错点 正确理解
1 直接 exp(x_i) 行不行？ 会溢出。必须减行 max，指数 ≤ 0 不溢出，分子分母同乘 exp(-max) 结果不变。
2 online softmax 那个 exp(m_prev - m) 是干嘛的？ rescale 因子：running max 从 m_prev 变到 m，之前按旧基准累的 d 整块乘它换算到新基准。
3 online 是 1-pass 吗？ 是 2-pass：pass 1 融合 max+sum，pass 2 仍要扫一遍做归一化。真 1-pass 是 FlashAttention（连归一化都融进算 O）。
4 online 一定更快？ 省的是访存（少读一趟 x），代价是每步多一次 expf 换算。x 装不进 cache 时才划算——这正是 FA 的场景。
5 reduce 用 shfl_down 还是 shfl_xor？ 用 shfl_xor：softmax 每个线程都要 max/sum，蝶式让全 lane 拿到结果；shfl_down 只有 lane 0 有。
6 GPU 上能直接套 CPU 的 online 递推？ 不能直接套。并行下各分块 (m,d) 同时算，要用带 rescale 的二元合并算子 d=d1·exp(m1-m)+d2·exp(m2-m)（结合律成立）才能并行归约。
7 一个 thread 一行行不行？ 浪费。行长 N 常达数千（vocab 15 万），要一 block 一行 + block reduce，block 内多线程 grid-stride 分摊。
